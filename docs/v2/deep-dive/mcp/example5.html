MCP Example 5 - Semantic Query (Zapper)

This example is the semantic-query runtime equivalent of Example 4.

Difference from Example 4:

  • Example 4 uses Database Knowledge Graph (DBKG) playbooks.
  • Example 5 uses semantic retrieval + schema graph + LLM AST + deterministic SQL compiler.
What this validates

This example validates your semantic pipeline end-to-end: retrieval, join path, AST generation, AST validation, SQL compile, SQL execute, and markdown-table summary.

{
  "mcp": {
    "action": "CALL_TOOL -> ANSWER",
    "toolCode": "db.semantic.query",
    "semantic": {
      "retrieval": {
        "selectedEntity": "DisconnectRequest",
        "candidateEntities": 1,
        "candidateTables": 1
      },
      "joinPath": {
        "baseTable": "zp_request",
        "edges": [
          "zp_request -> zp_disconnect_order",
          "zp_request -> zp_ui_data"
        ]
      },
      "astVersion": "v1",
      "astValid": true,
      "sqlCompiled": true,
      "sqlExecuted": true,
      "resultRows": 1
    }
  },
  "sql":
  select t1.zp_request_id as requestId, t1.account_id as accountId,
         t1.request_status as requestStatus, t1.requested_at as requestedAt,
         t0.don as disconnectOrderId, t0.downstream_status as downstreamStatus,
         t0.zp_disconnect_id as disconnectId, t2.zp_asr_team_notes as teamNotes
  from zp_disconnect_order as t0
  left outer
  join zp_request as t1
    on t0.zp_request_id = t1.zp_request_id
  left outer
  join zp_ui_data as t2
    on t1.zp_request_id = t2.zp_request_id
  where ((t1.request_status = :p1)
    and (t1.requested_at >= :p2))
  LIMIT :__limit
}

Generated SQL pattern (example)
SQL
select
  "t1"."zp_request_id" as "requestId", "t1"."account_id" as "accountId",
  "t1"."request_status" as "requestStatus", "t1"."requested_at" as "requestedAt",
  "t0"."don" as "disconnectOrderId", "t0"."downstream_status" as "downstreamStatus",
  "t0"."zp_disconnect_id" as "disconnectId", "t2"."zp_asr_team_notes" as "teamNotes"
from "zp_disconnect_order" as "t0"
left outer join "zp_request" as "t1"
  on "t0"."zp_request_id" = "t1"."zp_request_id"
left outer join "zp_ui_data" as "t2"
  on "t1"."zp_request_id" = "t2"."zp_request_id"
where (("t1"."request_status" = :p1) and ("t1"."requested_at" >= :p2))
limit :__limit
